VOL-170 plan

Project Framework Foundation

Build the staged Python project spine: stage handles, attached project tests, diagnostics, deterministic result writing, and docs.

Required execution skill: Use superpowers:subagent-driven-development or superpowers:executing-plans to implement task by task.

Goal: A user can define a staged Volt project, attach product-intent tests to stages, run through design/schematic/board stages, inspect diagnostics and test failures, and write a deterministic project bundle.

Architecture: Implement a pure-Python orchestration layer over existing Design, Schematic, and Board handles. Built-in stages are handle objects used as decorators, test registries, and result lookup keys. All EDA meaning remains inside existing kernel-owned models and validators.

Tech stack: Python 3.10+, pytest through CTest, existing volt._volt binding serializers.

from __future__ import annotations

from dataclasses import dataclass
from types import MappingProxyType
from typing import Callable, Mapping


BuildFunction = Callable[..., object]


@dataclass(frozen=True)
class _StageDefinition:
    name: str
    function: BuildFunction


@dataclass(frozen=True)
class _ProjectTest:
    name: str
    function: BuildFunction


class ProjectStage:
    """Stage handle for a Volt project workflow layer."""

    def __init__(self, project: "Project", name: str):
        self.project = project
        self.name = name
        self._function: BuildFunction | None = None
        self._tests: list[_ProjectTest] = []

    def __call__(self, function: BuildFunction) -> BuildFunction:
        if self._function is not None:
            raise RuntimeError(f"Project {self.project.name} {self.name} stage is already registered")
        self._function = function
        return function

    def test(self, function: BuildFunction) -> BuildFunction:
        self._tests.append(_ProjectTest(function.__name__, function))
        return function

    @property
    def registered(self) -> bool:
        return self._function is not None

    @property
    def tests(self) -> tuple[_ProjectTest, ...]:
        return tuple(self._tests)


class Project:
    """Structured Volt project entrypoint."""

    def __init__(
        self,
        name: str,
        *,
        version: str | None = None,
        description: str | None = None,
    ):
        if not isinstance(name, str):
            raise TypeError("Project name must be a string")
        if not name:
            raise ValueError("Project name must not be empty")
        self.name = name
        self.version = version
        self.description = description
        self.design = ProjectStage(self, "design")
        self.schematic = ProjectStage(self, "schematic")
        self.board = ProjectStage(self, "board")
        self._stage_order = (self.design, self.schematic, self.board)

    def build(self, function: BuildFunction) -> BuildFunction:
        return self._register_stage("build", function)

    def run(self):
        if not any(stage.registered for stage in self._stage_order):
            raise RuntimeError(f"Project {self.name} has no build stages")
        return ProjectResult(self)


class ProjectResult:
    """Result of one Volt project run."""

    def __init__(self, project: Project):
        self.project = project

    @property
    def ok(self) -> bool:
        return True

    @dataclass(frozen=True)
class StageResult:
    """Summary for one executed project stage."""

    name: str
    model_count: int


class Project:
    def run(self):
        return self._run_until(None)

    def run_through(self, stage: ProjectStage):
        if stage.project is not self:
            raise ValueError("Project stage belongs to a different project")
        return self._run_until(stage)

    def _run_until(self, stop_stage: ProjectStage | None):
        if not any(stage.registered for stage in self._stage_order):
            raise RuntimeError(f"Project {self.name} has no build stages")
        models: list[object] = []
        model_ids: set[int] = set()
        stages: list[StageResult] = []
        design = None
        for stage in self._stage_order:
            if not stage.registered:
                continue
            before = len(models)
            if stage is self.design:
                result = stage._function()
                self._collect_return(models, model_ids, result)
                design = _first_design(models)
            else:
                if design is None:
                    raise RuntimeError(f"Project {self.name} {stage.name} stage requires a design stage")
                result = stage._function(design)
                self._collect_return(models, model_ids, result)
            stages.append(StageResult(stage.name, len(models) - before))
            if stop_stage is stage:
                break
        return ProjectResult(self, stages=stages, models=models)

    def _collect_return(self, models: list[object], model_ids: set[int], value: object) -> None:
        if value is None:
            return
        if _is_known_model(value):
            if id(value) not in model_ids:
                models.append(value)
                model_ids.add(id(value))
            return
        if isinstance(value, (list, tuple)):
            for item in value:
                self._collect_return(models, model_ids, item)
            return
        raise TypeError("Project build stages must return Volt models or iterables of Volt models")

class ProjectDiagnostics:
    def __init__(self, diagnostics):
        self._diagnostics = tuple(diagnostics)

    def __iter__(self):
        return iter(self._diagnostics)

    def __len__(self):
        return len(self._diagnostics)

    def errors(self, *, stage: str | None = None):
        return tuple(
            diagnostic
            for diagnostic in self._diagnostics
            if diagnostic.severity == "error"
            and (stage is None or diagnostic.stage == stage)
        )

    @property
    def has_errors(self) -> bool:
        return bool(self.errors())


def _collect_default_diagnostics(models: tuple[object, ...]) -> ProjectDiagnostics:

    @dataclass(frozen=True)
class ProjectTestResult:
    stage: str
    name: str
    ok: bool
    message: str = ""


class DesignCheck:
    def __init__(self, design):
        self._design = design

    def net(self, name: str):
        return NetCheck(self._design, name)

    def no_connection(self, first: str, second: str) -> None:
        first_pins = set(self.net(first)._pin_labels())
        second_pins = set(self.net(second)._pin_labels())
        if first_pins & second_pins:
            raise AssertionError(f"{first} and {second} share connected pins")

    def ok(self, _message: str = "") -> None:
        return None


class NetCheck:
    def __init__(self, design, net_name: str):
        self._design = design
        self._net_name = net_name

    def connects(self, *pins: str) -> None:
        connected = set(self._pin_labels())
        missing = [pin for pin in pins if pin not in connected]
        if missing:
            raise AssertionError(
                f"Net {self._net_name} is missing expected pins: {', '.join(missing)}"
            )

    def _pin_labels(self) -> tuple[str, ...]:
        circuit = json.loads(self._design.to_json())
        net = next(item for item in circuit["nets"] if item["name"] == self._net_name)
        pins_by_id = {pin["id"]: pin for pin in circuit["pins"]}
        components_by_id = {component["id"]: component for component in circuit["components"]}
        pin_defs_by_id = {pin["id"]: pin for pin in circuit["pin_definitions"]}
        labels = []
        for pin_id in net["pins"]:
            pin = pins_by_id[pin_id]
            component = components_by_id[pin["component"]]
            pin_def = pin_defs_by_id[pin["definition"]]
            labels.append(f"{component['reference']}.{pin_def['name']}")
            labels.append(f"{component['reference']}.{pin_def['number']}")
        return tuple(labels)
